```js title="Example"
Input: s = "anagram", t = "nagaram"
Output: true

Input: s = "rat", t = "car"
Output: false
```

## Approach

1. **Sorting Method**: Sort both strings and compare them directly.
2. **Hash Table Method**: Count the frequency of each character in both strings and compare.
3. **Character Array**: Use a fixed-size array to count character frequencies.
4. **Implementation**: For the hash table method, increment counts for s and decrement for t, then check if all counts are zero.

## Solution

- Time Complexity: `O(n)`
  > Where n is the length of the strings. We traverse both strings once to count characters.
- Space Complexity: `O(1)`
  > We use a fixed-size array to store character counts (26 for lowercase letters).

```js tab="JavaScript"
/**
 * @param {string} s
 * @param {string} t
 * @return {boolean}
 */
var isAnagram = function(s, t) {
    if (s.length !== t.length) return false;
    
    const charCount = new Array(26).fill(0);
    
    // Count characters in s
    for (let char of s) {
        charCount[char.charCodeAt(0) - 'a'.charCodeAt(0)]++;
    }
    
    // Decrement counts for characters in t
    for (let char of t) {
        const index = char.charCodeAt(0) - 'a'.charCodeAt(0);
        charCount[index]--;
        
        // If count becomes negative, t has more of this character
        if (charCount[index] < 0) {
            return false;
        }
    }
    
    return true;
};
```

```java tab="Java"
class Solution {
    public boolean isAnagram(String s, String t) {
        if (s.length() != t.length()) return false;
        
        int[] charCount = new int[26];
        
        // Count characters in s
        for (char c : s.toCharArray()) {
            charCount[c - 'a']++;
        }
        
        // Decrement counts for characters in t
        for (char c : t.toCharArray()) {
            charCount[c - 'a']--;
            
            // If count becomes negative, t has more of this character
            if (charCount[c - 'a'] < 0) {
                return false;
            }
        }
        
        return true;
    }
}
```
