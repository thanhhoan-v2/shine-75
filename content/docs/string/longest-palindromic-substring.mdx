## Approach

1. **Expand Around Center**: For each character and each pair of characters, expand outward to find palindromes.
2. **Two Types of Centers**: Consider both single characters and pairs of characters as potential centers.
3. **Dynamic Programming**: Use a 2D array to track if substrings are palindromes.
4. **Manacher's Algorithm**: An optimized algorithm that runs in O(n) time.
5. **Implementation**: For each center, expand as long as characters match on both sides.

## Solution

- Time Complexity: `O(nÂ²)`
  > Where n is the length of the string. We check each character as a center and expand outward.
- Space Complexity: `O(1)`
  > We only use a constant amount of extra space.

```js tab="JavaScript"
/**
 * @param {string} s
 * @return {string}
 */
var longestPalindrome = function(s) {
    if (s.length < 2) return s;
    
    let start = 0;
    let maxLength = 1;
    
    // Helper function to expand around center
    const expandAroundCenter = (left, right) => {
        while (left >= 0 && right < s.length && s[left] === s[right]) {
            left--;
            right++;
        }
        return right - left - 1;
    };
    
    // Check each character and each pair as center
    for (let i = 0; i < s.length; i++) {
        // Expand around single character
        const len1 = expandAroundCenter(i, i);
        // Expand around pair of characters
        const len2 = expandAroundCenter(i, i + 1);
        
        const maxLen = Math.max(len1, len2);
        
        if (maxLen > maxLength) {
            start = i - Math.floor((maxLen - 1) / 2);
            maxLength = maxLen;
        }
    }
    
    return s.substring(start, start + maxLength);
};
```

```java tab="Java"
class Solution {
    public String longestPalindrome(String s) {
        if (s.length() < 2) return s;
        
        int start = 0;
        int maxLength = 1;
        
        // Helper function to expand around center
        private int expandAroundCenter(String s, int left, int right) {
            while (left >= 0 && right < s.length() && s.charAt(left) == s.charAt(right)) {
                left--;
                right++;
            }
            return right - left - 1;
        }
        
        // Check each character and each pair as center
        for (int i = 0; i < s.length(); i++) {
            // Expand around single character
            int len1 = expandAroundCenter(s, i, i);
            // Expand around pair of characters
            int len2 = expandAroundCenter(s, i, i + 1);
            
            int maxLen = Math.max(len1, len2);
            
            if (maxLen > maxLength) {
                start = i - (maxLen - 1) / 2;
                maxLength = maxLen;
            }
        }
        
        return s.substring(start, start + maxLength);
    }
}
```
